API DOC

	Todos los municipios : 

		url : http://cadizturismo/api/municipios

		response : [municipios]
						[lenguajes] ('es','en'...)
							['es'] ['markup'] : html
							['es'] ['viewBag'] : parametros, variables
	Un municipio :		

		url : http://cadizturismo/api/municipios/{municipio}
		params:
			{municipio} string (nombre del mismo en la url, de la peticion anterior, todos los municipios ejemplo : /municipios/barbate)

	Todas las comarcas :

		url : http://cadizturismo/api/comarcas
	
	Una comarca :

		url : http://cadizturismo/api/comarcas/{comarca}
		params:
			{comarca} string (title obtenido en la peticion anterior, todas las comarcas, ejemplo : /comarcas/La janda)

	Todas las playas

		url : http://cadizturismo/api/api/playas

	Una playa :

		url : http://cadizturismo/api/playas/{playa}
		params:
			{playa} string (title obtenido en la peticion anterior, todas las playas, ejemplo : /playas/Los lances )

	
	Todas las rutas

		url : http://cadizturismo/api/api/rutas
	
	Una Ruta :

		url : http://cadizturismo/api/rutas/{ruta}
		params:
			{ruta} string (title obtenido en la peticion anterior, todas las rutas, ejemplo : /rutas/Ruta de los pueblos blancos)

	Opciones rutas

		Categorias
			url : http://cadizturismo/api/api/rutas/opciones/categorias
		Ubicación
			url : http://cadizturismo/api/api/rutas/opciones/ubicacion

	Una ruta
	
		url : http://cadizturismo/api/rutas/{category}/{days}/{location}
		params:
			{category} string (value de las opciones categorias de rutas )
			{days} integer (numero de días)
			{location} string (value de las opciones ubicacion de rutas )


	Todas los eventos

		url : http://cadizturismo/api/api/eventos
	
	Un Evento:

		url : http://cadizturismo/api/eventos/{evento}
		params:
			{evento} string (title obtenido en la peticion anterior, todas los eventos, ejemplo : /eventos/XX Jornadas micológicas)

	Opciones eventos

		Categorias
			url : http://cadizturismo/api/api/eventos/opciones/categorias
		Ubicación
			url : http://cadizturismo/api/api/eventos/opciones/ubicacion

	Buscar un evento

		url : http://cadizturismo/api/eventos/{category}/{date_start}/{date_end}/{location}
		params:
			{category} string (value de las opciones categorias de eventos )
			{date_start} string (un string con el sig. formato :  07-02-2018 dia, mes, año)
			{date_end} string (un string con el sig. formato :  07-02-2018 dia, mes, año)
			{location} string (value de las opciones ubicacion de eventos )


	Opciones experiencias

		Intereses
			url : http://cadizturismo/api/experiencias/opciones/intereses
		Tipo de visita
			url : http://cadizturismo/api/experiencias/opciones/tipo-visita

	Una Experiencia:

		url : http://cadizturismo/api/experiencias/{experiencia}
		params:
			{experiencia} string (title obtenido en la peticion anterior, todas la experienciass, ejemplo : /experiencias/Rutas del vino de Cádiz)

	Buscar una experiencia

			url : http://cadizturismo/api/experiencias/{days}/{interest}/{tvisit}
			params :
				{days} integer (numero de días)
				{interest} string (value de las opciones intereses de experiencias )
				{tvisit} string (value de las opciones tipo de visita de experiencias )


	Páginas estaticas

			url : http://cadizturismo/api/estaticas


	Empresas 

		Tipos de empresas
			url : http://cadizturismo/api/empresas/opciones/tipos
			response : [tipos]

		Bucar empresas filtradas por tipo, municipios y cadana a buscar, sin paginación

			url : http://cadizturismo/api/empresas/{type_slug}/{municipalitiesSlugsString}/{searchString}
			
			params :
				{type_slug} String, example : alquiler-de-vehiculos
				{municipalitiesSlugsString} String, example: barbate_alcala-de-los-gazules_algar
				{searchString}: String, busca un string en el nombre y descripción de la empresa
				
			response : [empresas]

		Bucar empresas filtradas por tipo, municipios y cadana a buscar, con paginación

			url : http://cadizturismo/api/empresas/{type_slug}/{municipalitiesSlugsString}/{searchString}/{page}/{number}
			
			params :
				{type_slug} String, example : alquiler-de-vehiculos
				{municipalitiesSlugsString} String, example: barbate_alcala-de-los-gazules_algar
				{searchString}: String, busca un string en el nombre y descripción de la empresa
				{page} Number: número de pagína a mostrar
				{number} Number: número de items a mostrar por pagína




			response : [empresas]

				********************************************
				********************************************
				********************************************
				LYDIA
				********************************************
				********************************************
				********************************************
				{tlf2} String
				{email2} String
				{fax} String
				{cp} String

		Bucar empresas filtradas por tipo

			url : http://cadizturismo/api/empresas/por/tipo/{type_slug}
			
			params :
				{type_slug} String, example : alquiler-de-vehiculos
				
			response : [empresas]

		Una empresa por su nombre (name_slug)

			url : http://cadizturismo/api/empresa/{name_slug}/{lang}

			params :
				{name_slug} String, example : andalusia-tour-travel
				{lang} String, example : en
